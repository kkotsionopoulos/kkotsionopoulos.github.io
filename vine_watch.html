import os import requests import google.generativeai as genai def get_weather(): # Λήψη του κλειδιού από τις μεταβλητές περιβάλλοντος του GitHub Actions api_key = os.getenv('WEATHER_API_KEY') if not api_key: raise ValueError("Το WEATHER_API_KEY δεν βρέθηκε στις μεταβλητές περιβάλλοντος.") url = f"http://api.weatherapi.com/v1/current.json?key={api_key}&q=Tripoli" try: response = requests.get(url) data = response.json() # Έλεγχος αν το API επέστρεψε error (π.χ. Invalid Key) if 'error' in data: raise Exception(f"Σφάλμα από το WeatherAPI: {data['error']['message']} (Code: {data['error']['code']})") return data['current'] except Exception as e: raise Exception(f"Αποτυχία σύνδεσης με το WeatherAPI: {str(e)}") def main(): # 1. Λήψη δεδομένων καιρού weather_data = get_weather() print("Τα δεδομένα καιρού λήφθηκαν επιτυχώς.") # 2. Ρύθμιση του Gemini gemini_key = os.getenv('GEMINI_API_KEY') if not gemini_key: raise ValueError("Το GEMINI_API_KEY δεν βρέθηκε.") genai.configure(api_key=gemini_key) model = genai.GenerativeModel('gemini-pro') # 3. Δημιουργία περιεχομένου με βάση τον καιρό prompt = f"Ο καιρός στην Τρίπολη είναι: {weather_data['temp_c']} βαθμοί Κελσίου με {weather_data['condition']['text']}. Δώσε μια σύντομη γεωργική συμβουλή για την αμπελοκαλλιέργεια." response = model.generate_content(prompt) # 4. Αποθήκευση του αποτελέσματος στο αρχείο result.txt with open("result.txt", "w", encoding="utf-8") as f: f.write(response.text) print("Το αποτέλεσμα γράφτηκε στο result.txt") if __name__ == "__main__": main()
